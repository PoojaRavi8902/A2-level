# Program to find the sum of all odd integers in a 1-D array

def main():
    numbers = [12, 7, 5, 18, 21, 9, 14, 3]

    odd_sum = 0

    for n in numbers:
        if n % 2 != 0:    # check odd
            odd_sum += n

    print("Sum of all odd integers =", odd_sum)

Prime numbers — Program 1: Predefined array

1
Define helper function is_prime(n) to test primality.
2
If n ≤ 1 it's not prime; return False.
3
Check divisors from 2 up to sqrt(n). If any divide n, it's composite.
4
Main builds a list and prints numbers where is_prime returns True.
# Program to find all prime numbers in an array

def is_prime(n):
    if n <= 1:
        return False
    for i in range(2, int(n**2**0.5) + 1):
        if n % i == 0:
            return False
    return True


def main():
    numbers = [3, 8, 11, 15, 17, 22, 29, 31, 40, 41]

    print("Prime numbers in the array:")
    for num in numbers:
        if is_prime(num):
            print(num)

main()

# Linear search - simple version

def main():
    arr = [10, 14, 23, 5, 19, 7]
    key = 19

    found = False

    for i in range(len(arr)):
        if arr[i] == key:
            print("Found at position:", i)
            found = True
            break

    if not found:
        print("Not found")

main()

# Linear search using a function

def linear_search(arr, key):
    for i in range(len(arr)):
        if arr[i] == key:
            return i
    return -1


def main():
    data = [3, 9, 27, 6, 18]
    item = 6

    pos = linear_search(data, item)

    if pos == -1:
        print("Item not found")
    else:
        print("Item found at index:", pos)

main()

Bubble Sort — Program 1: Simple list version

1
Set up list arr to sort and get its length n.
2
Outer loop runs n times (each pass moves the next largest item to the end).
3
Inner loop compares adjacent elements up to the unsorted portion and swaps if needed.
4
After all passes, the list is sorted in ascending order.
# Bubble Sort - Simple Version

def main():
    arr = [23, 5, 12, 9, 30, 1]

    n = len(arr)

    for i in range(n):
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]

    print("Sorted list:", arr)

main()

Bubble Sort — Program 2: Efficient (Early Exit)

# Bubble Sort - Efficient Version with Early Exit

def bubble_sort(arr):
    n = len(arr)

    for i in range(n):
        swapped = False

        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True

        if not swapped:
            break


def main():
    data = [14, 3, 20, 7, 11, 5]
    bubble_sort(data)
    print("Sorted:", data)

main()

Bubble Sort — Program 3: While-loop version

1
Use a swapped boolean to control the while loop; keep iterating until no swaps.
2
Inner for-loop compares adjacent elements and swaps when necessary.
# Bubble Sort using while loop

def main():
    arr = [19, 4, 16, 8, 2]

    n = len(arr)
    swapped = True

    while swapped:
        swapped = False
        for i in range(n - 1):
            if arr[i] > arr[i + 1]:
                arr[i], arr[i + 1] = arr[i + 1], arr[i]
                swapped = True

    print("Sorted list:", arr)

main()

def bubble_sort(arr):

    for i in range(n):
        swapped = False
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:

# Bubble Sort with pass-by-pass display

def main():
    arr = [10, 7, 2, 15, 1]
    n = len(arr)

    for i in range(n):
        print("Pass", i + 1, "starting:", arr)
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
        print("Pass", i + 1, "after sorting:", arr, "\n")

main()
